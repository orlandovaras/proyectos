    <script>

        // Función para actualizar la hora

        function updateTime() {

            const now = new Date();

            const hours = String(now.getHours()).padStart(2, '0');

            const minutes = String(now.getMinutes()).padStart(2, '0');

            const seconds = String(now.getSeconds()).padStart(2, '0');

            

            document.getElementById('time').textContent = `${hours}:${minutes}:${seconds}`;

        }


        // Función para actualizar la fecha

        function updateDate() {

            const now = new Date();

            const options = { 

                weekday: 'long', 

                year: 'numeric', 

                month: 'long', 

                day: 'numeric' 

            };

            const dateStr = now.toLocaleDateString('es-CL', options);

            document.getElementById('date').textContent = dateStr.charAt(0).toUpperCase() + dateStr.slice(1);

        }


        // Función para determinar el icono del clima

        function getWeatherIcon(description) {

            const desc = description.toLowerCase();

            if (desc.includes('sol') || desc.includes('despej')) return '☀️';

            if (desc.includes('nub')) return '⛅';

            if (desc.includes('lluv')) return '🌧️';

            if (desc.includes('torment')) return '⛈️';

            if (desc.includes('niev')) return '❄️';

            return '🌤️';

        }


        // Inicializar

        updateTime();

        updateDate();

        setInterval(updateTime, 1000);

        setInterval(updateDate, 60000);


        // Actualizar clima basado en datos simulados de Santiago

        document.getElementById('weatherIcon').textContent = getWeatherIcon('Parcialmente nublado');
